%----------HIGHLIGHTS CUSTOM COMMANDS----------
\newcommand{\nametitle}[2]{%
    {\vspace*{-2.2cm}\Huge\bfseries\textls[150]{\textcolor{titlefontcolor}{\MakeUppercase{#1}}}}\\[0.4cm]
    {\color{titlefontcolor}\LARGE #2}
}

\newcommand{\profilepic}[1]{%
  \begin{center}
    {\vspace*{-3.5cm}\includegraphics[width=0.8\linewidth]{#1}}
  \end{center}
}

\newcommand{\contact}[2]{%
  \begin{tikzpicture}[baseline=(icon.base)]
    \node[draw=none, fill=accentcolor, circle, inner sep=2pt] (icon) {\color{black}#1};
    \node[anchor=west] at ([xshift=0.5em]icon.east) {\strut #2};
  \end{tikzpicture}%
}

\newcommand{\skill}[1]{%
  \tikz[baseline=(box.base)]{
    \node[
      draw=none,
      fill=accentcolor,
      rounded corners=10pt,
      inner sep=3pt,
      outer sep=10pt,
    ] (box) {\color{black}\strut #1};
  }%
}

\newcommand{\highlightlink}[2]{%
  \textcolor{tablecolor}{%
  \href{#1}{#2}%
  }%
}

\newcommand{\resumeHeadingListStart}{\begin{itemize}[topsep=0pt, partopsep=0pt, parsep=0pt, itemsep=10pt, leftmargin=-0.2cm, label={}]}
\newcommand{\resumeHeadingListEnd}{\end{itemize}}

\newcommand{\resumeNoBulletItemListStart}{\begin{itemize}[topsep=0pt, partopsep=0pt, parsep=0pt, itemsep=10pt, leftmargin=0pt, label={}]}
\newcommand{\resumeNoBulletItemListEnd}{\end{itemize}}

\newcommand{\resumeItemListStart}{\begin{itemize}[topsep=0pt, partopsep=0pt, parsep=0pt, itemsep=0pt, leftmargin=0.15in, label={\tiny$\bullet$}]}
\newcommand{\resumeItemListEnd}{\vspace{-15pt}\end{itemize}}

\newcommand{\resumeItem}[1]{\item\small{{#1}}}

\newcommand{\resumeExperience}[5]{%
    \item \begin{tabularx}{\linewidth}[t]{ c | X }
        \multirow[t]{3}{*}{\parbox[t]{\datewidth}{\centering #1 \par -- \par #2}} & \textbf{#3} \\
         & \textit{#4} \\
         & #5
    \end{tabularx}
}

\newcommand{\resumeEducation}[3]{%

}

\newcommand{\resumeTechstack}[2]{%

    #2
    }
}
